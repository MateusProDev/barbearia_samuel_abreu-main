// 🔐 Regras de Segurança do Firestore - VERSÃO COMPLETA
// Copie este código e cole no Firebase Console > Firestore > Regras

rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    
    // 👥 Coleção de usuários
    match /users/{userId} {
      // Permitir leitura para usuários autenticados
      allow read: if request.auth != null;
      // Permitir escrita apenas para admins ou o próprio usuário
      allow write: if request.auth != null && 
        (resource.data.isAdmin == true || request.auth.uid == userId);
    }
    
    // 📝 Conteúdo do site (siteContent)
    match /siteContent/{document} {
      // Permitir leitura para todos (site público)
      allow read: if true;
      // Permitir escrita apenas para usuários autenticados
      allow write: if request.auth != null;
    }
    
    // 🖼️ IMAGENS DO SITE - ESSENCIAL PARA FUNCIONAMENTO
    match /images/{imageId} {
      allow read: if true;
      allow write: if request.auth != null;
      allow delete: if request.auth != null;
    }
    
    // 📋 SEÇÕES DO SITE - NOVA COLEÇÃO
    match /sections/{sectionId} {
      allow read: if true;
      allow write: if request.auth != null;
    }
    
    // 📞 Informações de contato
    match /contact/{document} {
      // Permitir leitura para todos
      allow read: if true;
      // Permitir escrita apenas para usuários autenticados
      allow write: if request.auth != null;
    }
    
    // ✂️ Serviços
    match /services/{serviceId} {
      allow read: if true;
      allow write: if request.auth != null;
    }
    
    // 📸 Galeria
    match /gallery/{imageId} {
      allow read: if true;
      allow write: if request.auth != null;
    }
    
    // 👥 Equipe
    match /team/{memberId} {
      allow read: if true;
      allow write: if request.auth != null;
    }
    
    // ⚙️ Configurações gerais
    match /settings/{settingId} {
      allow read: if true;
      allow write: if request.auth != null;
    }
    
    // � Logs do sistema (opcional)
    match /logs/{logId} {
      allow read: if request.auth != null;
      allow write: if request.auth != null;
    }
    
    // 🔄 Migrações (para controle de versões)
    match /migrations/{migrationId} {
      allow read: if request.auth != null;
      allow write: if request.auth != null;
    }
  }
}
